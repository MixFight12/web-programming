import secrets
import string

def generate_password(length=16, use_symbols=True):
    chars = string.ascii_letters + string.digits
    
    if use_symbols:
        chars += string.punctuation
    password = ''.join(secrets.choice(chars) for _ in range(length))
    return password

print(generate_password(12))
